Java

import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int n = 0;
 
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        String s = fr.nextLine();
        String[] m = s.split("");
        for (int i = 0; i< m.length; i++){
            if (Integer.parseInt(m[i]) == 8) n+=2;
            else if (Integer.parseInt(m[i]) == 9) n++;
            else if (Integer.parseInt(m[i]) == 6) n++;
            else if (Integer.parseInt(m[i]) == 0) n++;
        }
        fw.println(n);
        fw.close();
    }
}
