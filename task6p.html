Java

import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int c = 0;
    int v = 0;
    int max;
    int min;
    byte b;
    int i1;
    int i2;
    int q = 0;
    int k = 1;
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        int a = fr.nextInt();
        int m[] = new int[a];
        for (int i = 0; i < a; i++){
            m[i] = fr.nextInt();
        }
        for (int i = 0; i < a; i++){
            for (int j = 0; j < a; j++){
                if (m[i] >= m[j]){
                    c++;
                }
                if (m[i] <= m[j]){
                    v++;
                }
            }
            if (m[i] > 0){
                q += m[i];
            }
            if (c == a){
                max = m[i];
 
            }
            else c = 0;
            if (v == a){
                min = m[i];
 
            }
            else v = 0;
        }
        for(int i = 0; i < a; i++){
            if (m[i] == max){
                i1 = i;
            }
            if (m[i] == min){
                i2 = i;
            }
        }
        for (int i = Math.min(i1, i2) + 1; i < Math.max(i1, i2); i++){
            k *= m[i];
        }
        if (m[i2] == max){
            k = 0;
        }
        fw.println(q + " " + k);
        fw.close();
       
    } 
}
