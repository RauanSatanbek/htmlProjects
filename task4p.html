Java

import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int c = 0;
    int v = 0;
    String s = "";
    String d = "";
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        int a = fr.nextInt();
        int m[] = new int [a];
        for (int i = 0; i < a; i++){
            m[i] = fr.nextInt();
        }
        for (int i = 0; i < a; i++){
            if (m[i] % 2 == 1){
                c++;
                s += m[i] + " ";
            }
            else {
                v++;
                d += m[i] + " ";
            }
        }
        if (c > v){
            fw.println(s + "\n" + d + "\nNO");
        }
        else fw.println(s + "\n" + d + "\nYES");
        fw.close();
    } 
}
