Java

import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int c = 0;
    int v = 0;
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        int a = fr.nextInt();
        int b = a/2;
        int m[] = new int[a];
        for (int i = 0; i < m.length; i++){
            m[i] = fr.nextInt();
        }
        Arrays.sort(m);
        for (int i = 0; i < m.length; i++){
            if (i < b){
                c += m[i];
            }
            else v +=m[i];
        }
        fw.println(v - c);
        fw.close();
        
    }
}
