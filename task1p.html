Java

import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int c = 0;
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        int m[] =new int[8];
        for (int i = 0; i < 8; i++){
            m[i] = fr.nextInt();
        }
        int x1 = m[0];
        int y1 = m[1];
         
        int x2 = m[2];
        int y2 = m[3];
         
        int x3 = m[4];
        int y3 = m[5];
         
        int x = m[6];
        int y = m[7];
         
        double A = Math.sqrt(Math.pow((x1 - x2), 2) + Math.pow((y1 - y2), 2));
        double B = Math.sqrt(Math.pow((x3 - x2), 2) + Math.pow((y3 - y2), 2));
        double C = Math.sqrt(Math.pow((x1 - x3), 2) + Math.pow((y1 - y3), 2));
        double p = (A + B + C)/2.0;
        double S = Math.sqrt(p * (p - A) * (p - B) * (p - C));
        //A, b
        double a = Math.sqrt(Math.pow((x1 - x), 2) + Math.pow((y1 - y), 2));
         
        //B, c
        double b = Math.sqrt(Math.pow((x - x2), 2) + Math.pow((y - y2), 2));
         
        // C, a
        double c = Math.sqrt(Math.pow((x3 - x), 2) + Math.pow((y3 - y), 2));
         
        double p1 = (A + b + a)/2.0;
        double p2 = (B + c + b)/2.0;
        double p3 = (C + a + c)/2.0;
         
        double S1 = Math.sqrt(p1 * (p1 - A) * (p1 - b) * (p1 - a));
        double S2 = Math.sqrt(p2 * (p2 - c) * (p2 - B) * (p2 - b));
        double S3 = Math.sqrt(p3 * (p3 - a) * (p3 - c) * (p3 - C));
         
        double S4 = S1 + S2 + S3;
         
        if (Math.round(S) == Math.round(S4)){
            fw.println("In");
        }
        else fw.print("Out");
         
        fw.close();
       
    } 
}
