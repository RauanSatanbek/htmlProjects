import java.io.*;
import java.util.*;
public class Main{
    public static void main(String args[]) throws IOException{
        new Main().run();
        }
    Scanner fr;
    PrintWriter fw;
    int c = 0;
    int p;
    private void run() throws IOException{
        fr = new Scanner(new File("input.txt"));
        fw = new PrintWriter(new File("output.txt"));
        int m[] =new int[6];
        for (int i = 0; i < 6; i++){
            m[i] = fr.nextInt();
        }
        int x1 = m[0];
        int y1 = m[1];
         
        int x2 = m[2];
        int y2 = m[3];
         
        int x3 = m[4];
        int y3 = m[5];
        
         
        double A = Math.sqrt(Math.pow((x1 - x2), 2) + Math.pow((y1 - y2), 2));
        double B = Math.sqrt(Math.pow((x3 - x2), 2) + Math.pow((y3 - y2), 2));
        double C = Math.sqrt(Math.pow((x1 - x3), 2) + Math.pow((y1 - y3), 2));
        double [] m1 = {A,B,C};
        for(int i = 0; i < 3; i++){
            for (int j = 0; j < 3; j++){
                if (m1[i] > m1[j]){
                    c++;
                }
                 
            }
            if (c == 2){
                p = i;
                break;
            }
            else {
                c = 0;
            }
        }
        if (p == 0){
            fw.println(((x1 + x2)-x3) + " " + ((y1 + y2)-y3));
             
        }
        else if (p == 1){
            fw.println(((x3 + x2)-x1) + " " + ((y3 + y2)-y1));
        }
        else{
            fw.println(((x3 + x1)-x2) + " " + ((y3 + y1)-y2));
        }
        fw.close();
       
    } 
}
